围棋的所有可能数361！的计算

计算结果：361! ≈ 1.4379*10^768

现在流行的围棋的棋盘是19行19列，也就是有19*19=361个落子之处，于是就想要计算一下围棋所有可能的棋局数量。

而第一子有361处可以落（只考虑可能的话，其实谁会开局去放个角落），第二子自然是360中可能，因此总的可能情况为：

361360359……2*1=361！

然后用程序大概计算它的大小。

简单计算

我用的第一种方法是每当结果快要超出double型的上限时，将结果除以10^100，并记录下来。这样代码比较方便，不用实现大数类，而结果可能会有些许误差。
代码如下：

// create by coffer
int calculate(int num)
{
    double product, step;
    int i, count;

    if (num == 0) // 0!=1
    {
        printf("%d! = %d\n", num, 1);
        return 0;
    }
    else if (num < 0) // 若传入负数则报错
    {
        printf("num must be right\n");
        return -1;
    }
    step = 1e100;
    product = 1;
    count = 0;

    for (i = 1; i <= num; i++)
    {
        if (product > step)
        {
            count++;
            product /= step;
        }
        product *= i;
    }

    if (count != 0)
        printf("%d! = %le*10^%d00\n", num, product, count);
    else
        printf("%d! = %le\n", num, product);

    return 0;
}

double型变量是64位，范围大约是-1.710(-308)～1.710(308)，于是我很简单的除以10^100，最后输出大概的量级。结果如下：

可以看到361！≈ 1.43*10^（700+68），你没看错，是10的768次这个量级（具体正确性没有考证，不过下面的另一个程序应该能够互相印证），正因如此，“千古无同局”这个说法太贴切了。
具体等于多少可以自己写一个大数类，或者用MATLAB计算出来。

而借着李世石和AlphaGo对局的机会，再来讨论一下人工智能的可怕。因为361！如此大，而现在的超级计算机也就达到十亿亿次/秒的量级，也就是10^17，想要在下围棋的4-5个小时内计算出所有可能完全是不可能的，既是对围棋的落子进行优化，也明显不可能从10^761降到10^17这个量级，因此今天AlphaGo是靠人工智能来战胜李世石的，这不废话么- -。其实这段话是看到朋友说“有什么好激动的，比运算速度有什么意思”，于是反思了一下，也唠叨一下~

说到计算速度，肯定不止这方面，上面也说了想计算棋局难度太大。通过机器学习才有可能，可是机器学习又是怎么实现的呢，我反正不得而知。看来人工智能方面的知识得好好补补课了。

利用大数类计算

这个大数类只实现了阶乘的计算，通用的网上应该能找到，其实我只是想自己写写代码啦。
big_number.h

// create by coffer
// big_number.h
#include <iostream>
using namespace std;

#define MAX_BYTE 10000 // 结果最大位数
#define LINE_FEED_NUM 50 // 结果每行显示位数

class Big_Number{
protected:
    int single_num[MAX_BYTE];
    int result[MAX_BYTE];
    int wait_for_count_num;
    int max_byte;
    int over_flow;

public:
    int count(int); // 计算结果的接口

protected:
    void print_result(); // 打印结果
    void init_num(); // 清空成员变量
    void calculate(); // 计算结果
} ;

big_number.cpp

// create by coffer
// big_number.cpp
#include "big_number.h"

// 计算结果的接口
int Big_Number::count(int n)
{
    wait_for_count_num = n;
    if (n < 0)
    {
        cout << "Number must be right.\n";
        return false;
    }
    else if (n == 0)
    {
        init_num();
        print_result();
    }
    else
    {
        init_num();
        calculate();
        if (over_flow == 1)
        {
            cout << "Number is too big.\n";
            return false;
        }
        print_result();
    }

    return true;
}

// 计算结果
void Big_Number::calculate()
{
    int step, i, j;
    int single_result;
    int carry, index_of_max_byte;
    int carry_carry_flag;

    index_of_max_byte = max_byte;
    for (step = 1; step <= wait_for_count_num; step++)
    {
        max_byte = index_of_max_byte;
        memcpy(single_num, result, sizeof(result));
        memset(result, 0, sizeof(result));
        for (i = 0; i <= max_byte; i++)
        {
            single_result = single_num[i] * step;
            result[i] += single_result % 10;
            if (result[i] >= 10)
            {
                carry_carry_flag = result[i] / 10;
                result[i] %= 10;
            }
            else
            {
                carry_carry_flag = 0;
            }
            carry = single_result / 10 + carry_carry_flag;

            j = 1;
            while (carry > 0)
            {
                if (i + j >= MAX_BYTE)
                {
                    over_flow = 1;
                    return;
                }
                result[i + j] += carry % 10;
                if (result[i + j] >= 10)
                {
                    carry_carry_flag = result[i + j] / 10;
                    result[i + j] %= 10;
                }
                else
                {
                    carry_carry_flag = 0;
                }
                carry = carry / 10 + carry_carry_flag;

                if ((i + j) > index_of_max_byte)
                {
                    index_of_max_byte = i + j;
                }

                j++;
            }
        }
    }

    max_byte = index_of_max_byte;
}

// 清空成员变量
void Big_Number::init_num()
{

    memset(result, 0, sizeof(result));
    memset(single_num, 0, sizeof(single_num));
    result[0] = 1;
    max_byte = 0; 
    over_flow = 0;

    return;
}

// 打印结果
void Big_Number::print_result()
{
    int i;
    int line_feed_count;

    cout << wait_for_count_num << "! = " << endl;

    i = max_byte;
    line_feed_count = 1;
    while (i>=0)
    {
        cout << result[i];
        i--;
        if (line_feed_count >= LINE_FEED_NUM)
        {
            cout << endl;
            line_feed_count = 1;
        }
        else
        {
            line_feed_count++;
        }
    }

    cout << endl;

    cout << "byte = " << max_byte + 1 << endl;

    return;
}
