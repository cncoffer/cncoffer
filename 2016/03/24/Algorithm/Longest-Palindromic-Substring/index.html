Longest Palindromic Substring

Question

Given a string S, find the longest palindromic substring in S. You may assume that the maximum length of S is 1000, and there exists one unique longest palindromic substring.
给你一个字符串S，找到S中的最长回文子串。S的最大长度是1000，并且其中有且只有一个最长回文子串。

Answer

暴力算法——三重循环

首先用笨办法实现一下，代码如下

// create by coffer
class Solution
{
public:
    string longestPalindrome(string s) {
        int i;
        int max_len, max_len_index;
        int length_count;
        //int str_end;
        int str_length = s.length();

        if (str_length <= 1)
        {
            return s; // 若长度小于1，则不用检查回文了
        }

        max_len = 1; // 若字符串不为空，回文子串长度最少是1
        max_len_index = 0;

        for (i = 0; i < str_length; i++) {
            length_count = max_len + 1; // 如果最长回文子串唯一，则应当从max_len+1开始检测

            while ( length_count + i < str_length) {
                if (true == is_Palindrome(s.substr(i, length_count))) {
                    max_len = length_count;
                    max_len_index = i;
                }
                length_count++;
            }
        }

        return s.substr(max_len_index, max_len);
    }

private:
    bool is_Palindrome(string s) {
        int i, end;
        end = s.length() - 1;

        if (end <= 0) {
            return true;
        }

        i = 0;
        while (i < (end-i)) {
            if (s[i] == s[end - i]) {
                i++;
            }
            else {
                return false;
            }
        }

        return true;
    }
public:
    Solution();
    ~Solution();
};

这个用的是最暴力的方法，时间复杂度为O(n^3)，太差了，因为提交超时了。
不过这也是意料之中的，3次方肯定是不能接受的。


暴力算法的改进

因为回文肯定是对称的，因此可以确定一个轴，从轴中间向两边对比是否是回文。
然后不停的移动轴来遍历整个字符串，时间复杂度降到O(n^2).
需要注意的一点是，轴有两种情况，对于字符串“abc”，轴可以位于b，也可以位于a与b之间，所以这两种情况都需要考虑。
代码如下：

#pragma once
#include <iostream>
#include <string>
using namespace std;
// create by coffer

class Solution
{
// "回文的对称轴"下面简称"轴"
#define MID_IS_EXIST 1 // 轴位于字符上的情况
#define MID_NOT_EXIST 0 // 轴位于两个字符之间的情况
public:
    Solution();
    ~Solution();
public:
    string longestPalindrome(string s) {
        int str_length = s.length();

        if (str_length <= 1) {
            return s;
        }

        int max_length, max_index;
        int length_count;
        int i;

        max_length = 1;
        max_index = 0;
        for (i = 0; i < str_length - 1; i++) { // 轴从第一个字符开始移动到最后一个字符,这里结束条件应该可以优化
            length_count = calcu_length(s, i, MID_IS_EXIST); // 轴位于第i个字符上
            if (length_count > max_length) {
                max_length = length_count;
                max_index = i - (length_count - 1) / 2;
            }
            length_count = calcu_length(s, i, MID_NOT_EXIST); // 轴位于第i个和第i+1个字符之间
            if (length_count > max_length) {
                max_length = length_count;
                max_index = i - length_count / 2 + 1;
            }
        }
        return s.substr(max_index, max_length);
    }

    int calcu_length(string s, int mid, int whether_mid_exist) {
        int front, end;
        int str_length = s.length();
        int length_count = 0;

        if (whether_mid_exist == MID_IS_EXIST) { // 对于轴位置的两种情况,做不同的处理
            front = mid - 1;
            end = mid + 1;
            length_count++;
        }
        else {
            front = mid;
            end = mid + 1;
        }

        // 从轴开始,向两边比较,知道找到不同的字符为止,返回计算得到的回文长度
        while ((front >= 0) && (end <= str_length - 1)) {
            if (s[front] == s[end]) {
                length_count += 2;
            }
            else {
                break;
            }
            front--;
            end++;
        }

        return length_count;
    }
};
