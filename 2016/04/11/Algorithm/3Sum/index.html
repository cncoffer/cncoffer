3Sum

Question

Given an array S of n integers, are there elements a, b, c in S such that a + b + c = 0? Find all unique triplets in the array which gives the sum of zero.

Note:

  • Elements in a triplet (a,b,c) must be in non-descending order. (ie, a ≤ b ≤ c)
  • The solution set must not contain duplicate triplets.
For example, given array S = {-1 0 1 2 -1 -4},
A solution set is:
(-1, 0, 1)
(-1, -1, 2)

翻译：
给定一有n个整数的数组S，S中是否有a,b,c三个元素满足a+b+c=0?找出所有和为0且不重复的三元组。
提示：

  • 给出的三元组(a,b,c)必须满足非递减的顺序，如a ≤ b ≤ c
  • 结果集合不包含重复的三元组

Answer

O(n^2logn)算法

因为要求和为0的三元组，因此当a和b确定后，可以直接从数组中查找是否有c，在排过序的数组中用二分查找复杂度是o(logn)，因此确立了第一种方法，先对数组排序，然后遍历所有a、b的组合，查找是否有c，若有c则记下结果，最后得到结果集合。
预期的时间复杂度为O(nlogn+n^2logn)=o(n^2*logn)
代码如下：

#pragma once
#include <iostream>
#include <vector>
//#include <cmath>
#include <algorithm>
using namespace std;

class Solution
{
public:
    Solution();
    ~Solution();
#define NOT_FOUND -1 // 表示查找失败
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> result;
        int i, j;
        int nums_size = nums.size();

        result.clear();
        if (nums_size <= 2) { // 如果size <= 2, 不可能有符合条件的三元组
            return result;
        }

        sort(nums.begin(), nums.end()); // sort默认是从小到大排序，用的快排

        int a, b, c;
        //int flag;
        i = 0; 
        a = nums[0] + 1;
        while (nums[i] < 0 && i < nums_size) {
            if (a != nums[i]) {
                a = nums[i];
                i++;
            }
            else {
                i++;
                continue;
            }
            j = nums_size - 1;
            b = nums[nums_size - 1] - 1;
            while (nums[j] > 0 && j >= 0) {
                if (b != nums[j]) {
                    b = nums[j];
                    j--;
                }
                else {
                    j--;
                    continue;
                }

                c = -a - b;
                if (c < a || c > b) { // 若c在[a,b]范围之外则会带来重复，需排除
                    continue;
                }
                else if (c == a || c == b) { // 若c与a或b相同，则查找c时需要找到两个c，否则其实是没有找到c
                    if (find_some_num(c, nums, 2) != NOT_FOUND) { // 查找数组中是否有两个c
                        result.push_back(order_result(a, b, c));
                    }
                }
                else {
                    if (find_some_num(c, nums, 1) != NOT_FOUND) { // 在数组中查找c
                        result.push_back(order_result(a, b, c));
                    }
                }
            }
        }
        if (find_some_num(0, nums, 3) != NOT_FOUND) { // 三数相同的情况仅(0,0,0)这一种
            result.push_back(order_result(0, 0, 0));
        }

        return result;
    }
private:
    int find_num(int c, vector<int>& nums) { // 用二分法，在排序过的nums中查找c
        int front, behind, mid;
        front = 0;
        behind = nums.size() - 1;
        mid = -1;

        while (front <= behind) {
            mid = (front + behind) / 2;
            if (nums[mid] == c) {
                return mid;
            }
            else if (nums[mid] < c) {
                front = mid + 1;
            }
            else {
                behind = mid - 1;
            }
        }

        return NOT_FOUND;
    }

    int find_some_num(int c, vector<int>& nums, int total) {
        int index = find_num(c, nums); // 首先调用二分法查找c
        if (total == 1) { // 若只需找到一个c，则返回结果
            return index;
        }

        int find_index, count;
        find_index = index - 1;
        count = 1;
        while (find_index >= 0) { // 若需要找到多个c，则从index-1开始向左查找c
            if (nums[find_index] != c) {
                break;
            }
            count++;
            if (count >= total) {
                return count;
            }
            find_index--;
        }
        find_index = index + 1;
        while (find_index < nums.size()) { // 若需要找到多个c，则从index+1开始向左查找c
            if (nums[find_index] != c) {
                break;
            }
            count++;
            if (count >= total) {
                return count;
            }
            find_index++;
        }
        return NOT_FOUND;
    }

    vector<int> order_result(int a, int b, int c) { // 对a、b、c排序，并返回vector变量
        vector<int> result;

        if (c < a) {
            result.push_back(c);
            result.push_back(a);
            result.push_back(b);
        }
        else if (c > b) {
            result.push_back(a);
            result.push_back(b);
            result.push_back(c);
        }
        else {
            result.push_back(a);
            result.push_back(c);
            result.push_back(b);
        }
        return result;
    }
};

O(n^2)算法

思路是先排序，然后确定a，在寻找满足a+b+c=0的b和c。
a可以从小到大递增，这是一重循环。
在寻找b和c时，b、c从两端向中间靠拢，可以规定b<=c（得到的结果自然是排好序的）。
另外在计算时，若a+b+c<0，则b递增，若>0,则c递减，这又是一重循环。
对比之下，我之前那样考虑各种特殊情况，还要对a、b、c进行排序的算法确实复杂的多。
代码如下：

#pragma once
#include <iostream>
#include <vector>
//#include <cmath>
#include <algorithm>
using namespace std;
class Solution2
{
public:
    Solution2();
    ~Solution2();
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        int num_size = nums.size();
        vector<vector<int>> result;

        result.clear();
        if (num_size < 3) { // 排除数字不满3个的特殊情况
            return result;
        }

        // 别忘记了排序
        sort(nums.begin(), nums.end());

        int sum;
        int a, b, c;
        for (a = 0; a < num_size - 1; a++) {
            if (a != 0 && nums[a] == nums[a-1]) { // 跳过相等的a
                continue;
            }
            b = a + 1;
            c = num_size - 1;
            while (b < c) {
                sum = nums[a] + nums[b] + nums[c];
                if (sum == 0) {
                    result.push_back(produce_vector(nums[a], nums[b], nums[c]));
                    b++;
                    c--;
                    while (b < c && nums[b] == nums[b - 1]) { // 跳过相等的b
                        b++;
                    }
                    while (c > b && nums[c] == nums[c + 1]) { // 跳过相等的c
                        c--;
                    }
                }
                else if (sum > 0) {
                    c--;
                    while (c > b && nums[c] == nums[c + 1]) { // 跳过相等的c
                        c--;
                    }
                }
                else {
                    b++;
                    while (b < c && nums[b] == nums[b - 1]) { // 跳过相等的b
                        b++;
                    }
                }
            }
        }
        return result;

    }
    vector<int> produce_vector(int a, int b, int c) {
        vector<int> result;
        result.push_back(a);
        result.push_back(b);
        result.push_back(c);
        return result;
    }
};

下面尝试对其进行优化。

优化的O(n^2)算法

对threeSum函数中的循环条件进行了修改，因为想要a+b+c=0，那最小数必须小于等于0，最大数必须大于等于0。这样可以减少a和c的遍历次数。
修改函数如下：

vector<vector<int>> threeSum(vector<int>& nums) {
    int num_size = nums.size();
    vector<vector<int>> result;

    result.clear();
    if (num_size < 3) { // 排除数字不满3个的特殊情况
        return result;
    }

    // 别忘记了排序
    sort(nums.begin(), nums.end());

    int sum;
    int a, b, c;
    for (a = 0; a < num_size - 1 && nums[a] <= 0; a++) { // 想要和为0，最小的nums[a]必须小于等于0
        if (a != 0 && nums[a] == nums[a-1]) { // 跳过相等的a
            continue;
        }
        b = a + 1;
        c = num_size - 1;
        while (b < c && nums[c] >= 0) { // 想要和为0，最大的nums[c]必须大于等于0
            sum = nums[a] + nums[b] + nums[c];
            if (sum == 0) {
                result.push_back(produce_vector(nums[a], nums[b], nums[c]));
                b++;
                c--;
                while (b < c && nums[b] == nums[b - 1]) { // 跳过相等的b
                    b++;
                }
                while (c > b && nums[c] == nums[c + 1]) { // 跳过相等的c
                    c--;
                }
            }
            else if (sum > 0) {
                c--;
                while (c > b && nums[c] == nums[c + 1]) { // 跳过相等的c
                    c--;
                }
            }
            else {
                b++;
                while (b < c && nums[b] == nums[b - 1]) { // 跳过相等的b
                    b++;
                }
            }
        }
    }
    return result;

}
